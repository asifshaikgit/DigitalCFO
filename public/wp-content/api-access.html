This page documents the REST(like) API for IDOS. Response format is JSON.

All API requests must have a Content-Type header set to application/json

All API requests must have a Accept header set to application/json

We use Token Authentication.
You have to pass the authentication token accessKey for every request.

You can generate an API key from here.

 String urlParameters = null;

 private HttpURLConnection setConnectionProperties(final String fetchUrl, final boolean isPost) throws IOException {
 URL url = new URL(fetchUrl);
 HttpURLConnection connection = (HttpURLConnection) url.openConnection();
 connection.setDoOutput(true);
 connection.setDoInput(true);
 connection.setInstanceFollowRedirects(false);
 if (isPost) {
 connection.setRequestMethod("POST");
 } else {
 connection.setRequestMethod("GET");
 }
 connection.setRequestProperty("Content-Type", "application/json");
 connection.setRequestProperty("charset", "utf-8");
 connection.setUseCaches(false);
 return connection;
 }

 private String getPostBody(final Map<String, Object> postBody) {
 String result = null;
 try {
 if (!postBody.isEmpty() && postBody.size() > 0) {
 ObjectNode node = Json.newObject();
 for (Map.Entry<String, Object> entry : postBody.entrySet()) {
 node.put(entry.getKey(), entry.getValue().toString());
 }
 result = Json.stringify(node);
 }
 } catch (Exception e) {
 e.printStackTrace();
 }
 return result;
 }

 public ObjectNode getResponse() {
 if (null == response) {
 response = Json.newObject();
 response.put("result", false);
 response.put("message", "Oops! Something went wrong.");
 }
 return response;
 }
